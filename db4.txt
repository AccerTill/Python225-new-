22. В записях с кодом товара: 1015, 1041, 1032, 1010 материал нейлон заменить на полиэстер если страна-производитель Китай.

UPDATE Waremain
SET Material = 'POLYESTER'
WHERE (ID = 1015 OR ID = 1041 OR ID = 1032 OR ID = 1010) AND Country = 'Китай' AND 
(Material = 'NYLON' OR Material = 'нейлон')


2)Variant (without material) 

UPDATE Waremain
SET Material = 'POLYESTER'
WHERE (ID = 1015 OR ID = 1041 OR ID = 1032 OR ID = 1010) 
AND Country = 'Китай'
